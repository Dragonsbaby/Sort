题目

add[1][2][3] + 4
输出10
add[10][20] + 30
输出60

上代码

js
const createPro = function(value = 0){
    return new Proxy({}, {
        get(target, prop){
            if(prop === Symbol.toPrimitive){
                return () => value;
            }
            return createPro(value + Number(prop));
        }
    });
};
let add = createPro();
add[1][2][3] + 4;
add[10][20] + 30;
